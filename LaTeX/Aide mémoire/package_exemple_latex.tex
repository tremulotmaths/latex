\begin{SideBySideExample}
    Si $\Delta > 0$ alors les deux racines sont :
    \begin{align*}
        x_1 &= \frac{-b - \sqrt\Delta}{2a}\\
        \shortintertext{et}
        x_2 &= \frac{-b + \sqrt\Delta}{2a}.
    \end{align*}
\end{SideBySideExample}
