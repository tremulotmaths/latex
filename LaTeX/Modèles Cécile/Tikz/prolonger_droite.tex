\section*{2ème méthode}

\begin{tikzpicture}
\coordinate (A) at (0,0) ;
\draw (A) node[above=5pt] {A} node {|};
\coordinate (B) at (2,0) ;
\draw (B) node[above=5pt] {B} node {|};
\draw[shorten < = -5mm, shorten > = -1cm] (A) -- (B);

\coordinate (E) at (3,1);
\fill (E) circle (1pt);
\draw (E) node[right] {\small $E$};
\coordinate (F) at ($(A)!(E)!(B)$);
\fill (F) circle (1pt);
\draw[dashed] (E) -- (F) ;
\end{tikzpicture}
