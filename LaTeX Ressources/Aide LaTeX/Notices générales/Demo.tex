En \LaTeX, on peut mettre du texte avec des maths \`{a} l'int\'{e}rieur, comme
$\int_{0}^{1}\sin x\,dx$ par exemple.

Mais aussi des \'{e}quations, num\'{e}rot\'{e}es ou non, ou avec un
num\'{e}ro ``custom'', et \'{e}ventuellement labellis\'{e}es pour rappeler
leur num\'{e}ro plus tard. Ici les \'{e}quations sont s\'{e}par\'{e}es ;
plus loin, elles seront group\'{e}es.
\[
\sum_{k=1}^{n}k=\frac{n(n+1)}{2}
\]
\begin{equation}
\sum_{k=1}^{n}k^{2}=\frac{n(n+1)(2n+1)}{6}
\end{equation}
\begin{equation}
\sum_{k=1}^{n}k^{3}=\frac{n^{2}(n+1)^{2}}{4}  \label{sommek3}
\end{equation}
\begin{equation}
\sum_{k=1}^{n}k^{4}=\frac{n\left( n+1\right) \left( 2n+1\right) \left(
3n^{2}+3n-1\right) }{30}
\tag{\ref{sommek3}$^{\prime }$}  \label{sommek4}
\end{equation}

L'\'{e}galit\'{e} (\ref{sommek3}) est courte, mais l'\'{e}galit\'{e} (\ref
{sommek4}) est longue.

Voici un th\'{e}or\`{e}me, suivi d'une preuve, d'un corollaire, d'une
proposition, entrecoup\'{e}s de remarques et d'exemples :

\begin{theorem}[th\'eor\`eme fondamental]
Quand le chien Till fait des cabrioles, il ne bave pas.
\end{theorem}

\proof
En effet, quand Till fait l'beau, il ne pleut pas.%
\endproof

Du texte pour faire du remplissage : quand un pitbull mord, il ne l\^{a}che
pas prise. Ceci explique pourquoi les mordus de pitbulls le restent,
malgr\'{e} tout le mal que font les pitbulls.

\begin{remark}
Tout ce qui pr\'{e}c\`{e}de est encore valable si le temps est mitig\'{e}.
\end{remark}

Encore du remplissage. Ce n'est pas toujours facile de trouver
l'inspiration, mais avec ce paragraphe, j'esp\`{e}re bien arriver \`{a} mes
fins : faire en sorte qu'il fasse plusieurs lignes... En g\'{e}n\'{e}ral,
quand les paragraphes ne font qu'une ligne, l'indentation a une forte
tendance \`{a} la mochet\'{e}.

\begin{corollary}
Quand il est au travail, Otto Mobile le tonnelier n'a pas \`{a} nettoyer ses
fringues toutes les cinq minutes.
\end{corollary}

\proof
On rappelle que Till est le chien d'Otto. On v\'{e}rifie ais\'{e}ment que
quand l'Otto Mobile fait des tonneaux avec le chien, celui-ci fait
forc\'{e}ment des cabrioles. Par cons\'{e}quent, les v\^{e}tements d'Otto ne
sont pas couverts de bave.%
\endproof

On n'est pas oblig\'{e} d'aller dans une sous-section pour mettre des choses
int\'{e}ressantes :

\begin{lemma}
\label{lempp}Soit $p$ un entier (strictement) positif. Alors il existe un
unique polyn\^{o}me $P_{p}\in \mathbf{C}[X]$ tel que pour tout entier
positif $n$,
\begin{equation}
\sum_{k=1}^{n}k^{p}=P_{p}(n)  \label{defpp}
\end{equation}
$P$ est \`{a} coefficients rationnels, de degr\'{e} $p+1$ et de coefficient
dominant $1/(p+1)$, v\'{e}rifie l'\'{e}galit\'{e} $P(0)=0$ et les
\'{e}quations fonctionnelles suivantes :
\begin{gather}
P_{p}(X+1)-P_{p}(X)=(X+1)^{p}  \label{ef1} \\
P_{p}(-X-1)=(-1)^{p+1}P_{p}(X)  \label{ef2}
\end{gather}
\end{lemma}

\begin{remark}
On peut aussi s'amuser \`{a} aligner les \'{e}quations sur leur signe
``\'{e}gale'' et m\^{e}me les \'{e}carter comme suit :
\end{remark}

\begin{align}
P_{p}(X+1)-P_{p}(X)& =(X+1)^{p}  \nonumber \\[0.4cm]
P_{p}(-X-1)& =(-1)^{p+1}P_{p}(X)
\end{align}

\proof[Preuve du lemme]
Si deux polyn\^{o}mes conviennent, ils co\"{i}ncident sur les entiers
positifs, donc sont \'{e}gaux~; ceci montre l'unicit\'{e}.

Cherchons un polyn\^{o}me $P_{p}=a_{0}+a_{1}X+\cdots +a_{p+1}X^{p+1}$ qui
v\'{e}rifie l'\'{e}galit\'{e} $P_{p}(0)=0$ et l'\'{e}quation fonctionnelle (%
\ref{ef1}). Pour cela, on doit avoir $a_{0}=0$ et :
\begin{align*}
\sum_{k=0}^{p}\mathsf{C}_{p}^{k}X^{k}& =\sum_{k=0}^{p+1}a_{k}\left(
\sum_{l=0}^{k}\mathsf{C}_{k}^{l}X^{l}-X^{k}\right) \\
& =\sum_{k=1}^{p+1}a_{k}\sum_{l=0}^{k-1}\mathsf{C}_{k}^{l}X^{l} \\[0.25cm]
\sum_{l=0}^{p}\mathsf{C}_{p}^{l}X^{l}& =\sum_{l=0}^{p}\left(
\sum_{k=l+1}^{p+1}a_{k}\mathsf{C}_{k}^{l}\right) X^{l}
\end{align*}
Cette derni\`{e}re \'{e}galit\'{e} donne lieu \`{a} un syst\`{e}me de $p+1$
\'{e}quations en les inconnues $a_{p+1},\ldots ,a_{1}$, \`{a} savoir $%
1=a_{p+1}.(p+1)$ (pour $l=p$), $\mathsf{C}_{p}^{p-1}=a_{p}\mathsf{C}%
_{p}^{p-1}+a_{p+1}\mathsf{C}_{p+1}^{p-1}$ (pour $l=p-1$), $\ldots $ , et $%
\mathsf{C}_{p}^{0}=a_{1}\mathsf{C}_{1}^{0}+a_{2}\mathsf{C}_{2}^{0}+\cdots
+a_{p+1}\mathsf{C}_{p+1}^{0}$ (pour $l=0$) ; on voit que ce syst\`{e}me se
r\'{e}sout successivement en $a_{p+1}$, $a_{p}$, $\ldots $ , $a_{1}$. Il
existe donc un polyn\^{o}me $P_{p}$ unique, de degr\'{e} au plus $p+1$,
v\'{e}rifiant $P_{p}(0)=0$ et l'\'{e}quation fonctionnelle (\ref{ef1}). Ce
polyn\^{o}me est \`{a} coefficients rationnels, de degr\'{e} $p+1$, et de
coefficient dominant $a_{p+1}=1/(p+1)$. L'\'{e}quation (\ref{ef1}) permet
ensuite de montrer par r\'{e}currence sur $n$ que pour tout entier positif $%
n $ on a : $P_{p}(n)=\sum_{k=1}^{n}k^{p}$. On trouve aussi : $%
P_{p}(-1+1)-P_{p}(-1)=(-1+1)^{p}=0$, donc $P_{p}(-1)=0$.

Soit $Q$ le polyn\^{o}me $(-1)^{p+1}P_{p}(-X-1)$. Il est de degr\'{e} $p+1$,
et $Q(0)=(-1)^{p+1}P_{p}(-1)=0$. On a \'{e}galement :
\begin{align*}
Q(X+1)-Q(X)& =(-1)^{p}\left[ P_{p}(-X-1)-P_{p}(-X-2)\right] \\
& =(-1)^{p}(-X-2+1)^{p} \\
& =(X+1)^{p}
\end{align*}
Donc $Q=P_{p}$, ce qui prouve l'\'{e}quation fonctionnelle (\ref{ef2}).%
\endproof

\begin{corollary}
$P_{p}(0)=P_{p}(-1)=0$ ; si $p$ est pair, $P_{p}(-1/2)=0$.
\end{corollary}

\proof
Tout a \'{e}t\'{e} vu, sauf l'annulation en $-1/2$, laquelle se d\'{e}duit
directement de l'\'{e}quation fonctionnelle (\ref{ef2}). On remarquera que
cette \'{e}quation signifie que le polyn\^{o}me $P_{p}(X-1/2)$ est pair
quand $p$ est impair, impair quand $p$ est pair.%
\endproof

\section{Comtesse.}

\begin{proposition}
Sissi n'\'{e}tait pas comtesse.
\end{proposition}

\proof
Si 6 comtesses scient 606 pens\'{e}es, c'est sans une pens\'{e}e pour chaque
comtesse. Or, Sissi aimait tous ses semblables (\cite{sis1}).%
\endproof

\section{Reine.}

\begin{proposition}
\label{propreine}Sissi n'\'{e}tait pas reine.
\end{proposition}

\proof
Sissi \'{e}tait marraine, pas reine. De plus, six sir\`{e}nes et deux pots
lisses, c'est tr\`{e}s bruyant.%
\endproof

\begin{remark}
Souvent un homme aimerait avoir six sir\`{e}nes \`{a} ses c\^{o}t\'{e}s ; il
doit cependant songer aux risques de noyade (\cite{ody}) :
\begin{equation}
\nu \int 6.sir\grave{e}ne=O\left( \frac{mera}{\acute{e}}\right)
\end{equation}
\end{remark}

Voici un petit plaisir en passant, qui illustre encore le lemme \ref{lempp}.
$:$%
\begin{equation}
\sum_{k=1}^{n}k^{6}=\frac{n\left( n+1\right) \left( 2n+1\right) \left(
3n^{4}+6n^{3}-3n+1\right) }{42}
\end{equation}

\section{Imp\'{e}ratrice.}

\subsection{Un p\`{e}re.}

Non, Sissi n'\'{e}tait pas un p\`{e}re. Voir la preuve de la proposition \ref
{propreine}.

\subsection{Actrice.}

Oui, c'est certain, Sissi \'{e}tait actrice. Voir tous les films avec Sissi (%
\cite{sis1}, \cite{sis2}, etc.).

\subsection{Conclusion.}

\begin{itemize}
\item  Dans certains films avec Sissi, elle \'{e}tait imp\'{e}ratrice (\cite
{sis2}). Dans d'autres (\cite{sis1}), non.

\item  Sissi n'\'{e}tait pas un p\`{e}re, mais \'{e}tait actrice.

\item  Sissi hait un titre moche.
\end{itemize}

Sissi \'{e}tait \`{a} demi imp\'{e}ratrice.

\def\indentedtrivlist#1{%
  \parsep\parskip
  \@nmbrlistfalse
  \@trivlist
  \labelwidth\z@
  \leftmargin\z@
  \linewidth\textwidth
  \advance\linewidth -#1
  \@totalleftmargin #1
  \parshape \@ne \@totalleftmargin \linewidth
  \let\@itemlabel\@empty
  \def\makelabel##1{##1}}
\def\endindentedtrivlist{%
  \if@inlabel
    \leavevmode
    \global \@inlabelfalse
  \fi
  \if@newlist
    \@noitemerr
    \global \@newlistfalse
  \fi
  \ifhmode\unskip \par\fi
  \if@noparlist \else
    \ifdim\lastskip >\z@
      \@tempskipa\lastskip \vskip -\lastskip
      \advance\@tempskipa\parskip \advance\@tempskipa -\@outerparskip
      \vskip\@tempskipa
    \fi
    \@endparenv
  \fi
}
