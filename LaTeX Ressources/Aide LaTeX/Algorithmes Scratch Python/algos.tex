\hspace*{2em}\textbf{Algorithme MaxCompatible(S)}

\hspace*{3em}\begin{algorithm}[H]
$S' \leftarrow 0$\;
Trier les activités de S par durée croissante\;
\Pour{$i$ de 1 à $\abs{S}$}{
\Si{l'activité $i$ est compatible avec les activités de $S'$}{
Ajouter l'activité $i$ à $S'$
}
}
\end{algorithm}
