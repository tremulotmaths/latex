\TitreActivite{vi.2}{Tableaux de signes}

\exo
\begin{enumerate}
    \item On donne le signe d'une expression $A(x)$ en fonction de $x$ dans le tableau ci-dessous. Résoudre :
    \[A(x) > 0 \qetq A(x) \leq 0.\]

    \begin{center}
        \begin{tikzpicture}
            \tkzTabInit[nocadre,espcl=2]{$x$/0.5,Signe de \\ $A(x)$/1}{$-\infty$,{$-2$},{$0,25$},$+\infty$}
            \tkzTabLine{,-,z,+,z,-}
        \end{tikzpicture}
    \end{center}

    \item À l'aide des tableaux de signe ci-dessous, résoudre :
    \[B(x) > 0 \qq C(x) = 0 \qq D(x) \leq 0.\]

    \begin{center}
        \begin{tikzpicture}
            \tkzTabInit[nocadre,espcl=1]{$x$/0.5,Signe de \\ $B(x)$/1}{$-\infty$,{$-3$},1,{$\sqrt 7$},$+\infty$}
            \tkzTabLine{,-,z,,,+,}
        \end{tikzpicture}
\hfill
        \begin{tikzpicture}
            \tkzTabInit[nocadre,espcl=1]{$x$/0.5,Signe de \\ $C(x)$/1}{$-\infty$,{$-3$},1,{$\sqrt 7$},$+\infty$}
            \tkzTabLine{,+,z,-,z,,+}
        \end{tikzpicture}
\hfill
        \begin{tikzpicture}
            \tkzTabInit[nocadre,espcl=1]{$x$/0.5,Signe de \\ $D(x)$/1}{$-\infty$,{$-3$},1,{$\sqrt 7$},$+\infty$}
            \tkzTabLine{,-,z,-,z,+,z,-}
        \end{tikzpicture}
    \end{center}
\end{enumerate}
