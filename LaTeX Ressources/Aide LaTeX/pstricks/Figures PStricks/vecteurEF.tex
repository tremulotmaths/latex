\pst@addfams{pst-vectorE}
%% [q1 x1 y1] [q2 x2 y2]
\define@key[psset]{pst-vectorE}{QXY}{\def\psk@vectorQXY{#1}}
\psset[pst-vectorE]{QXY=[1 -2 0][-1 2 0]}
\define@key[psset]{pst-vectorE}{K}{\def\psk@vectorEK{#1 }}
%% coefficient de longueur
\psset[pst-vectorE]{K=5}
\def\psvectorE{\pst@object{psvectorE}}
\def\psvectorE@i(#1){{% (x,y)
  \use@par
\pnode(! /QXY [\psk@vectorQXY] def % charges et positions
         /q1 QXY 0 get 0 get def
         /q2 QXY 1 get 0 get def
         /xA QXY 0 get 1 get def
         /yA QXY 0 get 2 get def
         /xB QXY 1 get 1 get def
         /yB QXY 1 get 2 get def
         /K \psk@vectorEK def
         0 0){datas}
\pst@getcoor{#1}\pst@tempA%
\pnode(!
    \pst@tempA
    /yP exch \pst@number\psyunit div def
    /xP exch \pst@number\psxunit div def
    /r1 xP xA sub dup mul yP yA sub dup mul add sqrt def
    /r2 xP xB sub dup mul yP yB sub dup mul add sqrt def
    /Ex q1 xP xA sub mul r1 3 exp div
        q2 xP xB sub mul r2 3 exp div
        add def
   /Ey q1 yP yA sub mul r1 3 exp div
       q2 yP yB sub mul r2 3 exp div
       add def
   /Angle Ey Ex atan def
   /NormeE Ex dup mul Ey dup mul add sqrt def
   /uEx Ex NormeE div K mul def
   /uEy Ey NormeE div K mul def
   0 0
    ){Champ}%
%\psdots[linecolor=red](!xA yA)(!xB yB)
%\psdot(!xP yP)
\psline{->}(!xP yP)(! xP Ex K mul add yP Ey K mul add)
}}
